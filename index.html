function createSunflower(x, z, delay) { const group = new THREE.Group(); // 1. TALLO CON CURVATURA (CatmullRomCurve3) // Creamos puntos aleatorios para generar una curva suave const points = []; const height = 4 + Math.random() * 2; const curveAmount = (Math.random() - 0.5) * 2; // Desviación lateral points.push(new THREE.Vector3(0, 0, 0)); points.push(new THREE.Vector3(curveAmount * 0.5, height * 0.5, curveAmount * 0.2)); points.push(new THREE.Vector3(curveAmount, height, 0)); const curve = new THREE.CatmullRomCurve3(points); const stemGeo = new THREE.TubeGeometry(curve, 20, 0.08, 8, false); const stemMat = new THREE.MeshStandardMaterial({ color: 0x2d5a27, roughness: 0.8 }); const stem = new THREE.Mesh(stemGeo, stemMat); group.add(stem); // 2. HOJAS ORGÁNICAS const leafShape = new THREE.Shape(); leafShape.moveTo(0, 0); leafShape.quadraticCurveTo(0.5, 0.5, 0, 1); leafShape.quadraticCurveTo(-0.5, 0.5, 0, 0); const leafGeo = new THREE.ShapeGeometry(leafShape); const leafMat = new THREE.MeshStandardMaterial({ color: 0x388e3c, side: THREE.DoubleSide }); for(let i = 0; i < 2; i++) { const leaf = new THREE.Mesh(leafGeo, leafMat); leaf.position.copy(curve.getPoint(0.4 + i * 0.2)); // Posicionadas en el tallo leaf.rotation.z = i % 2 === 0 ? 0.5 : -0.5; leaf.rotation.x = 1; leaf.scale.set(0.6, 0.6, 0.6); group.add(leaf); } // 3. CABEZA DEL GIRASOL (Inclinada según la curva del tallo) const headGroup = new THREE.Group(); const lastPoint = points[points.length - 1]; headGroup.position.set(lastPoint.x, lastPoint.y, lastPoint.z); // Orientar la cabeza hacia adelante ligeramente headGroup.rotation.x = 0.5; // Centro (Semillas) const centerGeo = new THREE.CylinderGeometry(0.6, 0.5, 0.2, 32); const centerMat = new THREE.MeshStandardMaterial({ color: 0x3b2519 }); const center = new THREE.Mesh(centerGeo, centerMat); center.rotation.x = Math.PI / 2; headGroup.add(center); // 4. PÉTALOS REALISTAS (Planos deformados) const petalShape = new THREE.Shape(); petalShape.moveTo(0, 0); petalShape.bezierCurveTo(0.2, 0.1, 0.3, 0.5, 0, 1.2); // Punta estilizada petalShape.bezierCurveTo(-0.3, 0.5, -0.2, 0.1, 0, 0); const petalGeo = new THREE.ShapeGeometry(petalShape); const petalMat = new THREE.MeshStandardMaterial({ color: 0xffeb3b, emissive: 0xffa000, emissiveIntensity: 0.2, side: THREE.DoubleSide }); const numPetals = 32; for (let i = 0; i < numPetals; i++) { const pGroup = new THREE.Group(); const petal = new THREE.Mesh(petalGeo, petalMat); const angle = (i / numPetals) * Math.PI * 2; pGroup.rotation.z = angle; petal.position.y = 0.5; // Distancia desde el centro petal.rotation.x = 0.2; // Inclinación natural hacia afuera pGroup.add(petal); headGroup.add(pGroup); } group.add(headGroup); group.position.set(x, 0, z); group.scale.set(0,0,0); scene.add(group); sunflowers.push({ mesh: group, delay: delay, curve: curve }); }
